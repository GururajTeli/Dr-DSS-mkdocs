def fetch_recipe_details(client,instance,project_key,engagement_type,final_list):
    """
    Fetches the recipe metadata from the DSS instance.

    Args:
        client (DSSClient): The DSS client instance.
        instance (str): The name of the DSS instance.
        project_key (str): The key of the project in DSS.
        engagement_type (str): The type of engagement for the project.
        final_list (list): List to store the final results.

    Returns:
        None
    """
    try:
        project = client.get_project(project_key)

        # Extract all the dataset information for the particular project
        recipes = project.list_recipes()

        for recipe in recipes:
            try:
                if not 'custom' in recipe.type.lower():
                    recipe_dict = dict()
                    input_dt = recipe['inputs']
                    output_dt = recipe['outputs']
                    # -------------------------------------------------------------------------------------------------
                    # DSS 1.0 fix during DSS 2.0 development

                    container_selection = 'NOT APPLICABLE'
                    # ---------------------------------------------------------------------------------------------------
                    if recipe.type == 'spark_sql_query':
                        engine = 'SPARK'
                    else :
                        recipe_st = recipe.to_recipe()
                        engine = recipe_st.get_status().get_selected_engine_details()['type']

                    # container details DONE
                    if recipe.type in ['python','r']:
                        if 'containerConf' in recipe['params']['containerSelection']:
                            container_selection = recipe['params']['containerSelection']['containerConf']
                        elif 'containerMode' in recipe['params']['containerSelection']:
                            if (recipe['params']['containerSelection']['containerMode'] == 'INHERIT') and ('containerConf' in project.get_settings().get_raw()['settings']['container']):
                                container_selection = project.get_settings().get_raw()['settings']['container']['containerConf']
                        else:
                            container_selection = 'NOT APPLICABLE'

                    # spark config details
                    if recipe.type in ['shaker']:
                        spark_config = recipe['params']['engineParams']['spark']['sparkConfig']['inheritConf']
                    elif recipe.type in ['spark_sql_query', 'pyspark']:
                        spark_config = recipe['params']['sparkConfig']['inheritConf']
                    elif recipe.type in ['sync', 'sampling']:
                        spark_config = recipe['params']['engineParams']['sparkSQL']['sparkConfig']['inheritConf']
                    else:
                        spark_config = ''

                    # SQL config details
                    if engine in ['SQL','S3_TO_SNOWFLAKE']:
                        container_selection = 'NOT APPLICABLE'
                        spark_config = 'NOT APPLICABLE'

                    # Extracting the json of only EXPORT recipes for the check of Hyper Upload Export Recipes
                    if recipe.type in ["export"] :
                        recipe_json = project.get_recipe(recipe.name).get_settings().get_recipe_raw_definition()['params']['exportParams']['originatingOptionId']
                    else:
                        recipe_json = " "

                    # Extracting the payload of only SparkSQL, SQL Query and SQL Script Recipes
                    if recipe.type in ["spark_sql_query","sql_query","sql_script"] :
                        recipe_content = project.get_recipe(recipe.name).get_settings().get_payload()
                    else :
                        recipe_content = " "

                    # Code for getting the last built of a recipe
                    if recipe.type not in ["export","prediction_training"]:
                        r_output_dataset = project.get_recipe(recipe.name).get_settings().get_recipe_outputs()

                        if "main_output" in r_output_dataset :
                            routputdataset_names = [item["ref"] for item in r_output_dataset["main_output"]["items"]][0]
                            dataset_object = project.get_dataset(routputdataset_names).get_info()

                            if "lastBuild" in dataset_object.get_raw() :
                                start_time = dataset_object.get_raw()["lastBuild"]["buildStartTime"]
                                end_time = dataset_object.get_raw()["lastBuild"]["buildEndTime"]
                                #end_time = datetime.fromtimestamp(dataset_object.get_raw()["lastBuild"]["buildEndTime"]/1000)
                                st = datetime.fromtimestamp(start_time/1000.0)
                                et = datetime.fromtimestamp(end_time/1000.0)
                                total_time = (et-st).total_seconds()
                            else :
                                start_time = 0
                                end_time = 0
                                total_time = 0

                        else :
                            routputdataset_names = [item["ref"] for item in r_output_dataset["main"]["items"]][0]
                            dataset_object = project.get_dataset(routputdataset_names).get_info()
                            if "lastBuild" in dataset_object.get_raw() :
                                start_time = dataset_object.get_raw()["lastBuild"]["buildStartTime"]
                                end_time = dataset_object.get_raw()["lastBuild"]["buildEndTime"]
                                #end_time = datetime.fromtimestamp(dataset_object.get_raw()["lastBuild"]["buildEndTime"]/1000)
                                st = datetime.fromtimestamp(start_time/1000.0)
                                et = datetime.fromtimestamp(end_time/1000.0)
                                total_time = (et-st).total_seconds()
                            else :
                                start_time = 0
                                end_time = 0
                                total_time = 0

                    else :
                        start_time = 0
                        end_time = 0
                        total_time = 0

                    # Format when only input of a recipe is present
                    if input_dt and not output_dt:
                        for i_data in input_dt['main']['items']:
                            final_list.append({'DSS_INSTANCE' : instance,
                                                    'RECIPE_NAME':recipe.name,
                                                    'PROJECT_KEY':project_key,
                                                    'ENGAGEMENT_TYPE':engagement_type,
                                                    'ENGINE':engine,
                                                    'RECIPE_TYPE':recipe.type,
                                                    'input':i_data['ref'],
                                                    'OUTPUT':'',
                                                    'SPARK_CONFIG':spark_config,
                                                    'CONTAINER': container_selection,
                                                    'TAGS': recipe.tags,
                                               'EXPORT_RECIPE_JSON':recipe_json,
                                               'RECIPE_CONTENT':recipe_content,
                                               'START_TIME':start_time,
                                               'END_TIME':end_time,
                                               'TOTAL_TIME':total_time
                                               })

                    # Format when only output of a recipe is present
                    elif output_dt and not input_dt:
                        for o_data in output_dt['main']['items']:
                            final_list.append({'DSS_INSTANCE' : instance,
                                                    'RECIPE_NAME':recipe.name,
                                                    'PROJECT_KEY':project_key,
                                                    'ENGAGEMENT_TYPE':engagement_type,
                                                    'ENGINE':engine,
                                                    'RECIPE_TYPE':recipe.type,
                                                    'INPUT': '',
                                                    'OUTPUT': o_data['ref'],
                                                    'SPARK_CONFIG':spark_config,
                                                    'CONTAINER': container_selection,
                                                    'TAGS': recipe.tags,
                                               'EXPORT_RECIPE_JSON':recipe_json,
                                               'RECIPE_CONTENT':recipe_content,
                                               'START_TIME':start_time,
                                               'END_TIME':end_time,
                                               'TOTAL_TIME':total_time
                                               })

                    # Format when both input and output of a recipe is present
                    elif input_dt and output_dt:
                        prdt = itertools.product(input_dt['main']['items'], output_dt['main']['items'])
                        for values in prdt:
                            final_list.append({'DSS_INSTANCE' : instance,
                                            'RECIPE_NAME':recipe.name,
                                            'PROJECT_KEY':project_key,
                                            'ENGAGEMENT_TYPE':engagement_type,
                                            'ENGINE':engine,
                                            'RECIPE_TYPE':recipe.type,
                                            'INPUT': values[0]['ref'],
                                            'OUTPUT': values[1]['ref'],
                                            'SPARK_CONFIG':spark_config,
                                            'CONTAINER': container_selection,
                                            'TAGS': recipe.tags,
                                               'EXPORT_RECIPE_JSON':recipe_json,
                                               'RECIPE_CONTENT':recipe_content,
                                               'START_TIME':start_time,
                                               'END_TIME':end_time,
                                               'TOTAL_TIME':total_time
                                               })
            except Exception as e:
                final_list.append({'DSS_INSTANCE' : instance,
                                    'RECIPE_NAME':recipe.name,
                                    'PROJECT_KEY':project_key,
                                    'ENGAGEMENT_TYPE':engagement_type,
                                    'ENGINE':'',
                                    'RECIPE_TYPE':recipe.type,
                                    'INPUT': '',
                                    'OUTPUT': '',
                                    'SPARK_CONFIG':'',
                                    'CONTAINER': e,
                                    'TAGS': recipe.tags,
                                   'EXPORT_RECIPE_JSON':" ",
                                   'RECIPE_CONTENT':" ",
                                   'START_TIME':0,
                                               'END_TIME':0,
                                               'TOTAL_TIME':0
                                   })

                print('EXCEPTION = {} for recipe {}'.format(str(e),recipe.name))
                continue
    except Exception as e:
        print (e,project_key,recipe.name,project_key)
