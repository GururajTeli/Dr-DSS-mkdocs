def fetch_project_details(client,instance,project_key,engagement_type,final_list,act_scenarios,job_log):
    """
    Fetches the project metadata from the DSS instance.

    Args:
        client (DSSClient): The DSS client instance.
        instance (str): The name of the DSS instance.
        project_key (str): The key of the project in DSS.
        engagement_type (str): The type of engagement for the project.
        final_list (list): List to store the final results.
        act_scenarios (list): List of active scenarios.
        job_log (str): Log file for jobs.

    Returns:
        None
    """
    try:
        project_obj = client.get_project(project_key)

        # Extracting the project permissions
        proj_permissions = project_obj.get_permissions()['permissions']

        # Extracting the project summary as metadata
        project_summary = project_obj.get_summary()  

        list_permissions_groups = []
        list_custom_permission = []

        project_settings = project_obj.get_settings()  # Project settings 
        project_pipeline_settings = project_settings.settings['settings']["flowBuildSettings"]

        # Extracting the Spark Pipeline
        spark_pipeline = project_pipeline_settings["mergeSparkPipelines"]

        # Extracting the SQL Pipeline
        sql_pipeline = project_pipeline_settings["mergeSqlPipelines"]

        # Extracting the flowzone names in the form of list
        zone_names = [zone.name for zone in project_obj.get_flow().list_zones()]

        project_timeline = project_obj.list_jobs()  # Extracting the jobs of the project
        successfull_jobs = []  # Storing the successfull jobs time

        # Iterate through each job in the project timeline
        for job in project_timeline:
            # Check if the job is completed ('stableState' is True) and successful ('state' is 'DONE')
            if job['stableState'] and job.get('state', '') == "DONE":
                # Extracting the end time of the job and storing in the list
                successfull_jobs.append(job["endTime"])

        if successfull_jobs:
            # Extracting the latest successfull job
            latestjobbuilt = max(successfull_jobs)
            latest_job_built = datetime.fromtimestamp(latestjobbuilt/1000)
        else:
            latest_job_built = None

        for item in proj_permissions:
            if 'group' in item:
                list_permissions_groups.append(item['group'])
            elif 'user' in item:
                list_custom_permission.append(item['user'])

        all_groups = list_permissions_groups
        all_custom = list_custom_permission

        last_actions = project_obj.get_timeline(item_count=1)['items'][0]
        first_act = project_obj.get_timeline(item_count=1000000)['items'][-1]
        if first_act['action']=='PROJECT_CREATE':
            created_at = first_act['time']
            created_by = first_act['user']
        else:
            created_at = 0
            created_by = ''
        final_list.append([instance
                               ,project_summary['projectKey']
                               ,project_summary['name']
                           ,engagement_type
                               ,project_summary['ownerLogin']
                               ,created_by
                               ,datetime.fromtimestamp(created_at/ 1000.0).strftime('%Y-%m-%d %H:%M:%S.%f')
                               ,(project_summary['tags'] if 'tags' in project_summary else '')
                               ,(project_summary['shortDesc'] if 'shortDesc' in project_summary else '')
                               ,len(project_obj.list_datasets(as_type='objects'))
                               ,len(project_obj.list_recipes(as_type='objects'))
                               ,len(project_obj.list_jupyter_notebooks(as_type='objects'))
                               ,len(project_obj.list_saved_models())
                               ,len(project_obj.list_scenarios(as_type='objects'))
                               ,len(project_obj.get_flow().list_zones())
                               ,all_groups
                               ,all_custom
                               ,last_actions['user']
                               ,last_actions['action']
                               ,last_actions['objectType']
                               ,last_actions['objectId']
                               ,last_actions['details']
                               ,datetime.fromtimestamp(last_actions["time"]/ 1000.0).strftime('%Y-%m-%d %H:%M:%S.%f')
                           # DSS 2.0 Addition
                               ,spark_pipeline
                           # DSS 2.0 Addition
                               ,sql_pipeline
                           # DSS 2.0 Addition
                               ,zone_names
                           # DSS 2.0 Addition
                               ,latest_job_built
                              ])
    except Exception as e:
        print(project_summary['projectKey'])
        print(e)

    try:
        act_scenarios.append([instance,
                              project_summary['projectKey'],
                              len(pd.DataFrame(project_obj.list_scenarios(as_type='listitems'))['active'].loc[lambda x : x==True])
                              ])
    except Exception as e:
        print("Exception from active scenario: "+str(e))

    try:
        jobs = project_obj.list_jobs()[0]
        job_log.append([instance,project_summary['projectKey']
                            ,jobs['def']['initiator']
                            ,datetime.fromtimestamp(jobs['startTime']/ 1000.0).strftime('%Y-%m-%d %H:%M:%S.%f')
                            ,datetime.fromtimestamp(jobs['endTime']/ 1000.0).strftime('%Y-%m-%d %H:%M:%S.%f')
                            ])
    except Exception as e:
        print("Exception from jobs: "+str(e))
