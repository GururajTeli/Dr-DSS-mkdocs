def fetch_notebook_details(client,instance,project_key,engagement_type,final_list):
    """
    Fetches the notebook metadata from the DSS instance.

    Args:
        client (DSSClient): The DSS client instance.
        instance (str): The name of the DSS instance.
        project_key (str): The key of the project in DSS.
        engagement_type (str): The type of engagement for the project.
        final_list (list): List to store the final results.

    Returns:
        None
    """

    try:
        # Define the timezone for IST and get today's date
        ist = pytz.timezone('Asia/Kolkata')
        now = datetime.now(ist)

        active_nb = []
        project = client.get_project(project_key)

        # Extract the Active Jupyter Notebook information for the particular project
        nb_list = project.list_jupyter_notebooks(active='True', as_type='listitems')

        # Extract all the Jupyter Notebook information for the particular project
        all_nb_list =  project.list_jupyter_notebooks(as_type='listitems')

        # Create the list of active notebook's names
        for nb in nb_list:
            active_nb.append(nb['name'])


        for nb in all_nb_list:
            is_active = 0

            # Flag the active notebooks in all_nb_list
            if nb['name'] in active_nb:
                is_active = 1

            # Find the hours a notebook has been active
            lst_modified_time = datetime.fromtimestamp(int(nb['lastModifiedOn'])/1000).replace(tzinfo=ist)
            diff = now - lst_modified_time
            diff_in_s = diff.total_seconds()
            active_time = divmod(diff_in_s, 3600)[0]

            # Format and append the required information
            final_list.append({'DSS_INSTANCE' : instance,
                                 'PROJECT_KEY':project_key,
                               'ENGAGEMENT_TYPE':engagement_type,
                                 'NOTEBOOK_NAME':nb['name'],
                                 'IS_ACTIVE' : is_active,
                                 'LAST_MODIFIED_ON': lst_modified_time,
                                 'ACTIVE_TIME': active_time,
                                 'LANGUAGE':nb['language'],
                                 'KERNEL':nb['kernelSpec'],
                                 'TAG':nb['tags']
                                })

    except Exception as e:
        print (e)
