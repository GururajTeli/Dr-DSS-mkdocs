def fetch_flowzone_details(client,instance,project_key,engagement_type,final_list):
    """
    Fetches the flowzone metadata from the DSS instance.

    Args:
        client (DSSClient): The DSS client instance.
        instance (str): The name of the DSS instance.
        project_key (str): The key of the project in DSS.
        engagement_type (str): The type of engagement for the project.
        final_list (list): List to store the final results.

    Returns:
        None
    """
    try :
        project = client.get_project(project_key)  # Project Object
        zones_id = [zone.id for zone in project.get_flow().list_zones()]  # Zones Metadata in the form of json
        datasets = project.list_datasets()  # Extracting list of datasets in the project
        no_flowzones = len(zones_id)  # Extracting count of flowzones present in the project

        zone_counts = {}

        for dataset in datasets: # Iterating through the datasets metadata
            dataset_object = project.get_dataset(dataset.name)  # Extracting dataset name
            zone = dataset_object.get_zone()  #C hecking for which zone id the dataset belongs to
            zone_name = zone.name  # Extracting the zone name from the zone id
            # Storing in the dictionary 
            if zone_name in zone_counts :
                zone_counts[zone_name] += 1
            else :
                zone_counts[zone_name] = 1

        for zoneid in zones_id:  # Iterating through the flowzone metadata
            flow_object = project.get_flow().get_zone(zoneid)
            jsonfile = flow_object.get_settings().get_raw()
            name = flow_object.get_settings().get_raw()["name"]  # Name of the flowzone
            tags = flow_object.get_settings().get_raw()["tags"]  # Tags of the flowzone

            final_list.append({
                    'DSS_INSTANCE':instance,
                    'PROJECT_NAME':project_key,
                    'ENGAGEMENT_TYPE':engagement_type,
                    'FLOWZONE_COUNT':len(zones_id),
                    'FLOWZONE_NAME':name,
                    'FLOWZONE_TAGS':tags,
                    # Extracting the description of the flowzone
                    'FLOWZONE_DESC':(flow_object.get_settings().get_raw()["description"] if "description" in jsonfile else " "),
                    # Extracting the short description of the flowzone
                    'FLOWZONE_SHORTDESC':(flow_object.get_settings().get_raw()["shortDesc"] if "shortDesc" in jsonfile else " "),  
                    'DATASET_COUNT' :(zone_counts.get(name) if name in zone_counts else 0)  # Dataet count in flowzone
                })

    except Exception as e:
        print("EXCEPTION = {} from project {}".format(str(e),str(project_key)))
