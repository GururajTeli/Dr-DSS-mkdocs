def fetch_dataset_details(client,instance,project_key,engagement_type,final_list):
    """
    Fetches the dataset metadata from the DSS instance.

    Args:
        client (DSSClient): The DSS client instance.
        instance (str): The name of the DSS instance.
        project_key (str): The key of the project in DSS.
        engagement_type (str): The type of engagement for the project.
        final_list (list): List to store the final results.

    Returns:
        None
    """
    project_obj = client.get_project(project_key)  # Project Object

    flow = project_obj.get_flow()  # Flow object of the project
    graph = flow.get_graph()  # Graph object of the project

    try:
        # Retrieve the list of datasets present in the project
        datasets = project_obj.list_datasets()

        # Get all recipe datasets from the graph (those with "COMPUTABLE_DATASET" type)
        all_recipe_datasets = [props["ref"] for name, props in graph.nodes.items() if props["type"] == "COMPUTABLE_DATASET"]

        # Create a list of dataset names in the project
        datasets_lst = [dataset.name for dataset in datasets]

        # Find datasets that are in the graph but not in the project's dataset list
        final_lst = [d_name for d_name in all_recipe_datasets if d_name not in datasets_lst]

        # Create a dictionary of managed folders with their IDs and types
        managed_folders = {folder['id']: folder['type'] for folder in project_obj.list_managed_folders()}

        # Iterate over each dataset to extract detailed metadata
        for dataset in datasets:
            if dataset["type"] == "S3":   # Check for only S3 datasets
                for name,props  in graph.nodes.items() :
                    if props["type"] == "COMPUTABLE_DATASET" and name == dataset.name:
                        input_items = props['predecessors'] # Check for predecessors
                        output_items = props['successors'] # Check for successors

                        recipename = props["predecessors"]
                        dataset_type = []

                        if len(recipename) != 0: # Check if the dataset has a recipe
                            recipe_settings = project_obj.get_recipe(recipename[0]).get_settings()
                            recipe_type = recipe_settings.get_recipe_raw_definition()["type"]

                            if recipe_type == "sync": # If the recipe type is "sync"
                                # Get input and output details of the recipe
                                recipeinputlist = recipe_settings.get_recipe_inputs()
                                rinputs = recipeinputlist.get('main',{}).get('items',[{}])[0].get('ref')
                                recipeoutputlist = recipe_settings.get_recipe_outputs()
                                routputs = recipeoutputlist.get('main',{}).get('items',[{}])[0].get('ref')

                                 # Retrieve and analyze dataset definitions for inputs and outputs
                                dataset_input_object = project_obj.get_dataset(rinputs)
                                dataset_input_definition = dataset_input_object.get_settings().get_raw()
                                dataset_output_object = project_obj.get_dataset(routputs)
                                dataset_output_definition = dataset_output_object.get_settings().get_raw()

                                dataset_type.append(dataset_input_definition["type"])

                                 # Set input and output formats based on dataset types
                                if dataset_input_definition["type"] == "S3" and dataset_output_definition["type"] == "S3":
                                    inputformat = [dataset_input_definition["formatType"]]
                                    outputformat = dataset_output_definition["formatType"]
                                else :
                                    inputformat = []
                                    outputformat = " "
                            else :
                                inputformat = []
                                outputformat = " "

                                # Process recipes other than "sync"
                                recipe_settings = project_obj.get_recipe(recipename[0]).get_settings()
                                recipe_type = recipe_settings.get_recipe_raw_definition()["type"]
                                recipe_input_list = recipe_settings.get_flat_input_refs()

                                if len(recipe_input_list) > 0:
                                    for r_input in recipe_input_list:
                                        if r_input not in list(managed_folders.keys()) and r_input not in final_lst:
                                            try:
                                                dataset_object = project_obj.get_dataset(r_input)
                                                dataset_definition = dataset_object.get_settings().get_raw()
                                                dataset_type.append(dataset_definition["type"])
                                                inputformat.append(dataset_definition["formatType"])
                                            except Exception as e:
                                                print("EXCEPTION = {} from project {}".format(str(e),str(project_key)))
                                        elif r_input in managed_folders:
                                            dataset_type.append(managed_folders[r_input])
                        else :
                            recipe_type = " "
                            dataset_type = []
                            inputformat = []
                            outputformat = " "
            else:
                recipe_type = " "
                dataset_type = []
                inputformat = []
                outputformat = " "
                input_items = []
                output_items = []

            # Append dataset metadata to the final_list
            final_list.append({'DSS_INSTANCE':instance
                               ,'PROJECT_NAME':project_key
                               ,'ENGAGEMENT_TYPE':engagement_type
                               ,'DATASET_NAME':dataset['name']
                               ,'DATASET_TYPE':dataset['type']
                               ,'DATASET_PARAMS':dataset['params']
                               ,'DATASET_CREATION':dataset['creationTag']
                               ,'DATASET_VERSION':dataset['versionTag']
                               ,'DATASET_TAGS':dataset["tags"]
                               ,'DATASET_CONNECTION':project_obj.get_dataset(dataset['name']).get_definition().get('params').get('connection')
                               ,'DATASET_INPUT_RECIPE_TYPE' :recipe_type
                               ,'DATASET_PREDECESSOR_TYPE' : list(set(dataset_type))
                               ,'PREDECESSOR_DATASET_FORMAT':list(set(inputformat))
                               ,'SUCCESSOR_DATASET_FORMAT':outputformat
                               ,'PREDECESSOR_ITEMS' :input_items
                               ,'SUCCESSOR_ITEMS' :output_items
                               ,"DATASET_USAGES" :project_obj.get_dataset(dataset['name']).get_usages()
                              })
    except Exception as e:
        # Handling specific exception
        if 'RemoteDisconnected' in str(e):
            # Filter out the entry that caused the error from final_list
            final_list[:] = [item for item in final_list if not (item['DSS_INSTANCE'] == instance and item['PROJECT_NAME'] == project_key)]
        print("EXCEPTION = {} from project {}".format(str(e), str(project_key)))
