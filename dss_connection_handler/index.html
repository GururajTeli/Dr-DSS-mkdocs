def connect_and_fetch_metadata(instance, dss_credentials, project_key, engagement_type, metadata_type, final_list, act_scenarios, job_log):
    """
    Establishes a connection to a DSS instance and calls the appropriate metadata extraction functions based on the metadata_type.

    Examples:
        >>> from dss_metadata_utils import connect_and_fetch_metadata 
        >>> dss_credentials = {"instance1": ["http://dss-instance1.com", "api-key-1"]}
        >>> final_list = []
        >>> act_scenarios = []
        >>> job_log = []
        >>> metadata_type: The type of metadata to retrieve. Accepted values can be:
                            'project_metadata'
                            'dataset_metadata'
                            'flowzone_metadata'
                            'notebook_metadata'
                            'recipe_metadata'
                            'scenario_metadata'
                            'shared_dataset_metadata'
        >>> connect_and_fetch_metadata(instance="instance1",
                                        dss_credentials=dss_credentials,
                                        project_key="PROJECT_123",
                                        engagement_type="type1",
                                        metadata_type="project_metadata",
                                        final_list=final_list,
                                        act_scenarios=act_scenarios,
                                        job_log=job_log)

    Args:
        instance (str): The name of the DSS instance.
        dss_credentials (dict): Dictionary containing DSS instance credentials.
        project_key (str): The key of the project in DSS.
        engagement_type (str): The type of engagement for the project.
        metadata_type (str): The type of metadata to retrieve.
        final_list (list): List to store the final results.
        act_scenarios (list): List of active scenarios.
        job_log (list): Log file for jobs.

    Returns:
        None
    """

    # Retrieve the list of available DSS instances
    dss_instances = list(dss_credentials.keys())

    # Only proceed if the selected instance is available in the the dss_instances list to avoid any errors
    if instance in dss_instances:
        host = dss_credentials[instance][0] # Host connection URL
        api_key = dss_credentials[instance][1] # API Key to be used to authenticate the connection
        client = DSSClient(host,api_key) # Establishing the connection to the DSS client
        client._session.verify = False # To remove the Certificate fail error

        if metadata_type.lower() == 'project_metadata':
            # Function call - project_metadata
            fetch_project_details(client, instance, project_key, engagement_type, final_list, act_scenarios, job_log)

        elif metadata_type.lower() == 'dataset_metadata':
            # Function call - dataset_metadata
            fetch_dataset_details(client,instance,project_key,engagement_type,final_list)

        elif metadata_type.lower() == 'flowzone_metadata':
            # Function call - flowzone_metadata
            fetch_flowzone_details(client,instance,project_key,engagement_type,final_list)

        elif metadata_type.lower() == 'notebook_metadata':
            # Function call - notebook_metadata
            fetch_notebook_details(client,instance,project_key,engagement_type,final_list)

        elif metadata_type.lower() == 'recipe_metadata':
            # Function call - recipe_metadata
             fetch_recipe_details(client,instance,project_key,engagement_type,final_list)

        elif metadata_type.lower() == 'scenario_metadata':
            # Function call - scenario_metadata
             fetch_scenario_details(client,instance,project_key,engagement_type,final_list)

        elif metadata_type.lower() == 'shared_dataset_metadata':
            # Function call - shared_dataset_metadata
             fetch_shared_dataset_details(client,instance,final_list)
