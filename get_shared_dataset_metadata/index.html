def fetch_shared_dataset_details(client,instance,final_list):
    """
    Fetches the shared dataset metadata from the DSS instance.

    Args:
        client (DSSClient): The DSS client instance.
        instance (str): The name of the DSS instance.
        final_list (list): List to store the final results.

    Returns:
        None
    """

    dss_projects = client.list_project_keys()
    for project in dss_projects:
        try:
            project_obj = client.get_project(project)
            raw_settings = project_obj.get_settings().get_raw()

            # Get the list of shared datasets in a project
            for exposed_obj in raw_settings["exposedObjects"]["objects"]:
                try:
                    connection = project_obj.get_dataset(exposed_obj["localName"]).get_definition().get('params').get('connection')

                    # For each shared dataset, get the target project
                    for rule in exposed_obj["rules"]:
                        target_project = rule["targetProject"]

                        # Format and append the required information
                        final_list.append({'DSS_INSTANCE' : instance,
                                         'PROJECT_KEY': project,
                                         'SHARED_DATASET':exposed_obj["localName"],
                                         'DATASET_CONNECTION': connection,
                                         'TARGET_PROJECT': target_project
                                        })
                except Exception as e:
                    print("EXCEPTION = {} from project {}".format(str(e),str(project)))
                    continue
        except Exception as e:
            print(e)
