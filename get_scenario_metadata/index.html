def fetch_scenario_details(client,instance,project_key,engagement_type,final_list):
    """
    Fetches the scenario metadata from the DSS instance.

    Args:
        client (DSSClient): The DSS client instance.
        instance (str): The name of the DSS instance.
        project_key (str): The key of the project in DSS.
        engagement_type (str): The type of engagement for the project.
        final_list (list): List to store the final results.

    Returns:
        list: The updated final_list containing scenario metadata.
    """

    # Retrieve the project object using the provided project key
    project_object = client.get_project(project_key)

    # Retrieve the list of scenarios associated with the project
    scenarios = project_object.list_scenarios()

    # Initialize counters and lists for active scenarios and those with active triggers
    active_scenarios_count = 0
    active_scenarios = []

    scenarios_with_active_triggers = []
    scenarios_with_active_triggers_count = 0

    # Check if there are scenarios in the project
    if len(scenarios) != 0 :
        # Get the total number of scenarios
        total_scenarios = len(scenarios)

        # Identify active scenarios and count them
        for scenario_info in scenarios:
            if scenario_info["active"] is True :
                active_scenarios_count += 1
                active_scenarios.append(scenario_info["id"])

        # Check each active scenario for active triggers
        for scenario_id in active_scenarios :
            scenario = project_object.get_scenario(scenario_id)
            scenario_settings = scenario.get_settings()
            scenario_raw = scenario_settings.get_raw()
            triggers = scenario_raw.get('triggers', [])

            # Count scenarios with active triggers
            if triggers != []:
                for indv in triggers :
                    if indv["active"] == True :
                        scenarios_with_active_triggers.append(scenario_id)
                        scenarios_with_active_triggers_count += 1

    else:
        # If no scenarios are present, set the total scenarios count to zero
        total_scenarios = len(scenarios)

    # Append scenario metadata to the final_list
    final_list.append({'DSS_INSTANCE':instance,
                               'PROJECT_NAME':project_key,
                               'ENGAGEMENT_TYPE':engagement_type,
                               'TOTAL_SCENARIOS':total_scenarios,
                               'ACTIVE_SCENARIOS':active_scenarios,
                               'ACTIVE_SCENARIOS_COUNT' : active_scenarios_count,
                               'SCENARIOS_WITH_ACTIVE_TRIGGERS' :scenarios_with_active_triggers,
                               'SCENARIOS_WITH_ACTIVE_TRIGGERS_COUNT' :scenarios_with_active_triggers_count,
                               'SCENARIO_NAMES' :[item["name"] for item in scenarios]
                              })
    return final_list
